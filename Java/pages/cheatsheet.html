Control Flow Patterns

Javaif / else / switch / loops
// if-else
if (x > 0) { ... }
else if (x == 0) { ... }
else { ... }

// Ternary
String label = (x > 0) ? "positive" : "non-positive";

// Switch (classic)
switch (day) {
    case "MON": case "TUE": result = "weekday"; break;
    default: result = "weekend";
}

// Switch expression (Java 14+)
String result = switch (day) {
    case "MON", "TUE", "WED", "THU", "FRI" -> "weekday";
    case "SAT", "SUN" -> "weekend";
    default -> throw new IllegalArgumentException();
};

// For loop
for (int i = 0; i < n; i++) { ... }

// Enhanced for
for (String item : list) { ... }

// While
while (condition) { ... }

// Do-while
do { ... } while (condition);

// break / continue / labeled break
outer:
for (int i = 0; i < 10; i++) {
    for (int j = 0; j < 10; j++) {
        if (j == 5) continue outer;
        if (i == 8) break outer;
    }
}

Object-Oriented Essentials

JavaClass, interface, inheritance, record
// ── CLASS ──────────────────────────
public class Animal {
    private String name;
    protected int age;

    public Animal(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public String getName() { return name; }

    @Override
    public String toString() {
        return "Animal{name='%s'}".formatted(name);
    }
}

// ── INHERITANCE ───────────────────
public class Dog extends Animal {
    private String breed;

    public Dog(String name, int age, String breed) {
        super(name, age);          // Must call parent constructor
        this.breed = breed;
    }

    @Override
    public String toString() {
        return "Dog{name='%s', breed='%s'}".formatted(getName(), breed);
    }
}

// ── INTERFACE ─────────────────────
public interface Speakable {
    String speak();                        // Abstract
    default String whisper() {             // Default method
        return speak().toLowerCase();
    }
    static Speakable silent() {            // Static factory
        return () -> "";
    }
}

// ── ABSTRACT CLASS ────────────────
public abstract class Shape {
    abstract double area();
    double perimeter() { return 0; }       // Concrete method
}

// ── RECORD (Java 16+) ────────────
public record Point(int x, int y) {
    // Compact constructor for validation
    public Point {
        if (x < 0 || y < 0) throw new IllegalArgumentException();
    }
}

// ── ENUM ──────────────────────────
public enum Direction {
    NORTH, SOUTH, EAST, WEST;
    public Direction opposite() {
        return switch (this) {
            case NORTH -> SOUTH;
            case SOUTH -> NORTH;
            case EAST  -> WEST;
            case WEST  -> EAST;
        };
    }
}

// ── SEALED CLASS (Java 17+) ──────
public sealed class Result permits Success, Failure {}
public final class Success extends Result {}
public final class Failure extends Result {}

Streams API

Java
// Creating streams
Stream.of("a", "b", "c");
list.stream();
Arrays.stream(array);
IntStream.range(0, 10);          // 0..9
IntStream.rangeClosed(1, 10);    // 1..10
Stream.iterate(0, n -> n + 2);   // 0, 2, 4, 6, ...
Stream.generate(Math::random);

// Intermediate (lazy)
.filter(x -> x > 5)
.map(String::toUpperCase)
.flatMap(Collection::stream)
.distinct()
.sorted()
.sorted(Comparator.comparing(Person::age))
.peek(System.out::println)       // Debug only
.limit(10)
.skip(5)

// Terminal (triggers execution)
.forEach(System.out::println)
.collect(Collectors.toList())
.collect(Collectors.toSet())
.collect(Collectors.joining(", "))
.collect(Collectors.groupingBy(Person::city))
.collect(Collectors.partitioningBy(n -> n > 0))
.collect(Collectors.toMap(Person::name, Person::age))
.reduce(0, Integer::sum)
.count()
.findFirst()                     // Optional<T>
.anyMatch(x -> x > 10)
.allMatch(x -> x > 0)
.min(Comparator.naturalOrder())
.toArray(String[]::new)

opt.orElse("default");
opt.orElseThrow();

Exceptions & Error Handling

Java
// try-catch-finally
try {
    riskyOperation();
} catch (IOException | SQLException e) {   // Multi-catch
    logger.error("Failed", e);
} finally {
    cleanup();   // Always runs
}

// try-with-resources (AutoCloseable)
try (var reader = new BufferedReader(new FileReader("file.txt"));
     var writer = new BufferedWriter(new FileWriter("out.txt"))) {
    // Resources auto-closed at end
}

// Custom exception
public class OrderException extends RuntimeException {
    private final String orderId;
    public OrderException(String orderId, String message) {
        super(message);
        this.orderId = orderId;
    }
    public String getOrderId() { return orderId; }
}

// Throwing
throw new IllegalArgumentException("Invalid input: " + value);
throw new OrderException("ORD-123", "Payment failed");
ℹ️
Checked vs Unchecked
Checked (extends Exception): Must catch or declare — IOException, SQLException
Unchecked (extends RuntimeException): No obligation — NullPointerException, IllegalArgumentException

File I/O Quick Patterns

Java
// Read entire file (Java 11+)
String content = Files.readString(Path.of("file.txt"));
List<String> lines = Files.readAllLines(Path.of("file.txt"));

// Write file
Files.writeString(Path.of("out.txt"), "Hello");
Files.write(Path.of("out.txt"), lines);

// Stream lines (lazy, good for large files)
try (Stream<String> stream = Files.lines(Path.of("big.txt"))) {
    stream.filter(line -> line.contains("ERROR"))
          .forEach(System.out::println);
}

// Walk directory tree
try (Stream<Path> paths = Files.walk(Path.of("src"))) {
    paths.filter(p -> p.toString().endsWith(".java"))
         .forEach(System.out::println);
}

// Path operations
Path p = Path.of("src", "main", "App.java");
p.getFileName();       // App.java
p.getParent();         // src/main
p.toAbsolutePath();
Files.exists(p);
Files.isDirectory(p);
Files.createDirectories(Path.of("a/b/c"));

Concurrency Essentials

Java
// Thread creation
Thread t = new Thread(() -> System.out.println("Running"));
t.start();

// ExecutorService (preferred)
ExecutorService pool = Executors.newFixedThreadPool(4);
Future<Integer> future = pool.submit(() -> compute());
int result = future.get();   // Blocking
pool.shutdown();

// CompletableFuture
CompletableFuture.supplyAsync(() -> fetchData())
    .thenApply(data -> process(data))
    .thenAccept(result -> save(result))
    .exceptionally(ex -> { log(ex); return null; });

// Synchronized
synchronized (lock) {
    sharedResource.update();
}

// Concurrent collections
ConcurrentHashMap<String, Integer> map = new ConcurrentHashMap<>();
CopyOnWriteArrayList<String> list = new CopyOnWriteArrayList<>();
BlockingQueue<Task> queue = new LinkedBlockingQueue<>();

// Virtual threads (Java 21+)
Thread.startVirtualThread(() -> handleRequest());
try (var executor = Executors.newVirtualThreadPerTaskExecutor()) {
    executor.submit(() -> task());
}
